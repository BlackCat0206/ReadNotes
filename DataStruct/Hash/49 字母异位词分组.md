# 49_字母异位词分组

[字母异位词分组](https://leetcode.cn/problems/group-anagrams/)

## 



### X_题解

* 使用排序后的`str`作为Key

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <unordered_map>
#include <algorithm>

using namespace std;

vector<vector<string>> groupAnagrams(vector<string>& strs) {
    // 哈希表：key=排序后的字符串，value=异位词列表
    unordered_map<string, vector<string>> mp;
    
    for (string& s : strs) {
        string key = s;          // 拷贝原字符串
        sort(key.begin(), key.end());  // 排序得到唯一标识
        mp[key].push_back(s);   // 归入对应分组
    }
    
    // 提取哈希表中的所有分组作为结果
    vector<vector<string>> res;
    for (auto& pair : mp) {
        res.push_back(pair.second);
    }
    
    return res;
}

// 测试示例
int main() {
    vector<string> strs = {"eat", "tea", "tan", "ate", "nat", "bat"};
    vector<vector<string>> result = groupAnagrams(strs);
    
    // 输出结果
    for (auto& group : result) {
        for (auto& s : group) {
            cout << s << " ";
        }
        cout << endl;
    }
    // 输出：
    // eat tea ate 
    // tan nat 
    // bat 
    return 0;
}
```

* 使用字符统计作为key

```cpp
class Solution {
public:
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
        // 将输入的字符串按照一定的规律排序
        // 将排序的结果进行比较，若出现相同的内容，则添加对应坐标
        unordered_map<string, vector<string>> umResult;

        for (auto &s : strs)
        {
            vector<int> count(26, 0);
            for (char c : s)
            {
                count[c - 'a']++;
            }

            string key;
            for (int num : count)
            {
                key += '#' + num;
            }
            umResult[key].push_back(s);
        }

        // 提取结果
        vector<vector<string>> res;
        for (auto &pair : umResult)
        {
            res.push_back(pair.second);
        }

        return res;
    }
};
```
